@startuml
title Fluxo 6.3 — Recuperação de Senha (Diagrama de Sequência)

actor Usuario
participant "Navegador" as Browser
participant "Flask/Auth Blueprint" as AuthBP
participant "AuthService" as Service
participant "SMTP Server" as SMTP
participant "AuditService" as Audit
database "PostgreSQL" as DB

== Solicitação de recuperação ==

Usuario -> Browser : Acessa "Esqueci minha senha"
Browser -> AuthBP : POST /recuperar (cpf)

AuthBP -> Service : solicitar_recuperacao(cpf)

Service -> Service : validar_cpf(cpf)
alt CPF inválido
    Service --> AuthBP : erro("CPF inválido")
    AuthBP --> Browser : resposta erro
else CPF válido
    Service -> DB : SELECT usuario por cpf
    DB --> Service : usuario / null

    alt Usuário não encontrado
        Service --> AuthBP : resposta genérica
        AuthBP --> Browser : "Se existir, e-mail enviado"
    else Usuário encontrado
        Service -> Service : gerar_token()
        Service -> DB : INSERT token
        Service -> SMTP : enviar_email(link com token)
        SMTP --> Service : ok

        Service --> AuthBP : sucesso
        AuthBP --> Browser : "E-mail enviado"
    end
end

== Redefinição de senha ==

Usuario -> Browser : Acessa link com token
Browser -> AuthBP : GET /redefinir?token=XXX

AuthBP -> Service : validar_token(token)
Service -> DB : SELECT token
DB --> Service : dados do token

alt Token inválido ou expirado
    Service --> AuthBP : erro("Token inválido")
    AuthBP --> Browser : exibir erro
else Token válido
    AuthBP --> Browser : exibir formulário nova senha

    Usuario -> Browser : Envia nova senha
    Browser -> AuthBP : POST /redefinir (token, senha)

    AuthBP -> Service : redefinir_senha(token, senha)

    Service -> DB : UPDATE senha (hash)
    Service -> DB : invalidar_token
    Service -> Service : encerrar_sessoes

    Service -> Audit : registrar(operação)
    Audit -> DB : INSERT audit_log

    Service --> AuthBP : sucesso
    AuthBP --> Browser : redirect login
end
