@startuml
title RF-003 — Logout de Usuário (Diagrama de Atividade)

start

:Usuário autenticado acessa opção "Sair";

:Enviar requisição de logout;

if (Sessão ativa?) then (Não)
    :Redirecionar para tela de login;
    stop
else (Sim)
endif

:Encerrar sessão (Flask-Login);

:Invalidar dados da sessão;

:Registrar operação no audit_log;

:Redirecionar para tela de login;

stop

@enduml
