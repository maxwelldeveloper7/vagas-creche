@startuml
title RF-001 — Autenticar Usuário (Diagrama de Atividade)

start

:Exibir tela de login;

:Usuário informa CPF e senha;

:Enviar formulário;

if (CPF válido?) then (Não)
    :Exibir mensagem "CPF inválido";
    stop
else (Sim)
endif

if (Usuário existe?) then (Não)
    :Exibir mensagem "Usuário não encontrado";
    stop
else (Sim)
endif

if (Conta ativa?) then (Não)
    :Exibir mensagem "Conta bloqueada/inativa";
    stop
else (Sim)
endif

if (Senha correta?) then (Não)
    :Incrementar tentativas de login;
    
    if (Tentativas >= limite?) then (Sim)
        :Bloquear conta;
        :Registrar no audit_log;
        :Exibir mensagem "Conta bloqueada";
        stop
    else (Não)
        :Exibir mensagem "Credenciais inválidas";
        stop
    endif

else (Sim)
endif

:Resetar tentativas de login;

:Registrar login no audit_log;

:Criar sessão (Flask-Login);

:Redirecionar para dashboard;

stop

@enduml
