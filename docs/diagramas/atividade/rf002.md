@startuml
title RF-002 — Recuperação de Senha (Diagrama de Atividade)

start

:Exibir tela "Esqueci minha senha";

:Usuário informa CPF;

:Enviar solicitação;

if (CPF válido?) then (Não)
    :Exibir mensagem "CPF inválido";
    stop
else (Sim)
endif

if (Usuário existe?) then (Não)
    :Exibir mensagem "Usuário não encontrado";
    stop
else (Sim)
endif

:Gerar token de recuperação;

:Definir expiração do token;

:Armazenar token no banco;

:Enviar e-mail com link de redefinição;

:Exibir mensagem "E-mail enviado";

stop

partition "Redefinição de Senha" {

    :Usuário acessa link com token;

    if (Token válido?) then (Não)
        :Exibir mensagem "Token inválido ou expirado";
        stop
    else (Sim)
    endif

    :Exibir formulário de nova senha;

    :Usuário informa nova senha;

    if (Senha atende critérios?) then (Não)
        :Exibir mensagem "Senha inválida";
        stop
    else (Sim)
    endif

    :Atualizar senha (hash);

    :Invalidar token;

    :Encerrar sessões ativas;

    :Registrar operação no audit_log;

    :Redirecionar para login com sucesso;

}

stop

@enduml
