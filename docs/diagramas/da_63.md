@startuml
title Fluxo 6.3 — Recuperação de Senha (Diagrama de Atividade)

start

:Usuário acessa opção "Esqueci minha senha";

:Informa CPF e envia solicitação;

:Validar formato do CPF;
if (CPF válido?) then (Sim)

  :Buscar usuário no banco;
  if (Usuário encontrado?) then (Sim)

    :Gerar token de recuperação;
    :Armazenar token com expiração;
    :Enviar e-mail com link de redefinição;

    :Exibir mensagem "E-mail enviado";

    :Usuário acessa link com token;

    :Validar token;
    if (Token válido e não expirado?) then (Sim)

      :Exibir formulário de nova senha;

      :Usuário informa nova senha;

      :Atualizar hash da senha no banco;
      :Invalidar token utilizado;
      :Encerrar sessões ativas;

      :Registrar operação no audit_log;

      :Redirecionar para login com mensagem de sucesso;

    else (Não)
      :Exibir erro "Token inválido ou expirado";
    endif

  else (Não)
    :Exibir mensagem genérica (não revela existência do usuário);
  endif

else (Não)
  :Exibir erro "CPF inválido";
endif

stop

@enduml
