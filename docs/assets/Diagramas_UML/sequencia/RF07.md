@startuml

actor "Secretário/Diretor" as Usuario
participant "Interface Web" as UI
participant "Servidor Flask" as Sistema
database "PostgreSQL" as BD

Usuario -> UI : Preenche formulário de cadastro da criança
UI -> Sistema : enviarDadosCrianca()

Sistema -> BD : salvarCrianca()
BD --> Sistema : confirmação

Sistema -> UI : exibirMensagem("Deseja cadastrar outra criança?")

Usuario -> UI : Seleciona opção (Sim)

UI -> Sistema : solicitarNovoCadastro()

Sistema -> UI : perguntarMesmoRelacionamento()

Usuario -> UI : responde (Sim ou Não)

alt Mesmo relacionamento = SIM
    Sistema -> UI : preencherNomePaiMaeAutomaticamente()
else Mesmo relacionamento = NÃO
    Sistema -> UI : abrirFormularioVazio()
end

Usuario -> UI : revisa/edita dados
UI -> Sistema : enviarDadosNovaCrianca()

Sistema -> BD : salvarCrianca()
BD --> Sistema : confirmação

Sistema -> UI : exibirMensagem("Cadastro realizado com sucesso")

@enduml
