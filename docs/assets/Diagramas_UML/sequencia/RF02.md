@startuml

actor Administrador

participant "Interface Web\n(Formulário HTML)" as UI
participant "Servidor Flask\n(Controlador)" as Controller
participant "Camada de Persistência\n(Model)" as Model
database "PostgreSQL\nBanco de Dados" as DB

Administrador -> UI : Acessar tela de cadastro de unidade

UI -> Controller : GET /unidades/nova

Controller -> UI : Renderizar formulário de cadastro

Administrador -> UI : Preencher dados da unidade\n(nome, endereço)

UI -> Controller : POST /unidades\n(dados do formulário)

Controller -> Controller : Validar dados

alt Dados inválidos

Controller -> UI : Retornar erro de validação

else Dados válidos

Controller -> Model : criar_unidade(dados)

Model -> DB : INSERT INTO unidades_escolares

DB --> Model : confirmação

Model --> Controller : unidade criada

Controller -> UI : Redirecionar para lista de unidades

end

@enduml
