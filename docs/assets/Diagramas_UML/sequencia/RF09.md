@startuml

actor "Operador\n(Diretor/Secretário)" as Operador
participant "Interface Web\n(Flask Template)" as Interface
participant "Servidor Flask\n(Controlador)" as Servidor
participant "Serviço de Validação" as Validacao
database "Banco de Dados\nPostgreSQL" as BD
participant "Gerador de Comprovante\n(PDF)" as PDF

Operador -> Interface : Inicia nova inscrição

Interface -> Servidor : POST /verificar-cpf-crianca

Servidor -> BD : consultar CPF da criança
BD --> Servidor : resultado da consulta

alt CPF já possui inscrição
    Servidor -> Interface : informar duplicidade
    Interface -> Operador : exibir mensagem de erro
else CPF disponível
    Servidor -> Interface : abrir formulário de inscrição
end

Operador -> Interface : preencher dados do responsável
Operador -> Interface : preencher dados da criança

Interface -> Servidor : POST /salvar-inscricao

Servidor -> Validacao : validar dados obrigatórios
Validacao --> Servidor : dados válidos

Servidor -> BD : inserir responsável
BD --> Servidor : id_responsavel

Servidor -> BD : inserir criança
BD --> Servidor : id_crianca

Servidor -> BD : inserir inscrição
BD --> Servidor : numero_inscricao

Servidor -> PDF : gerar comprovante
PDF --> Servidor : arquivo PDF

Servidor -> Interface : exibir confirmação

Interface -> Operador : mostrar comprovante disponível

@enduml
